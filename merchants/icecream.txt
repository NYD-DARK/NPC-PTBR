-	script	Ice Cream Maker::IceCreamer	-1,{
	mes "[Ice Cream Maker]";
	mes "Sorvete fresco feito com neve de Lutie!";
	mes "Aproveite agora, năo ficará ŕ venda por muito tempo!";
	mes "^3355FF100 Zenys^000000 por Sorvete,";
	mes "Sorvete!";
	next;
	if(select("Me dę um Sorvete!:Cancelar negociaçăo") == 2) {
		mes "Tem certeza de que năo quer nenhum?";
		mes "Năo vou vendę-los por muito tempo,";
		mes "e assim que acabarem,";
		mes "năo haverá mais!!!";
		close;
	}
	mes "[Ice Cream Maker]";
	mes "Sorvete fresco feito com neve de Lutie!";
	mes "Todos querem nosso delicioso sorvete,";
	mes "mas temos uma quantidade limitada,";
	mes "entăo vocę só pode comprar 5 por vez!!";
	next;
	while(true) {
		input .@input;
		if(.@input < 1) {
			mes "[Ice Cream Maker]";
			mes "Se vocę năo quer comprar nada,";
			mes "poderia, por favor, deixar o próximo cliente";
			mes "fazer uma compra?";
			mes "Obrigado.";
			close;
		}
		if(.@input > 5) {
			mes "[Ice Cream Maker]";
			mes "Ai";
			mes "Vocę espera demais.";
			mes "Caro cliente,";
			mes "Se vocę comer mais de 5 Sorvetes,";
			mes "Talvez vocę tenha que fazer muitas visitas";
			mes "ao banheiro esta noite.";
			next;
			continue;
		}
		break;
	}
	if(Zeny<.@input*100) {
		mes "[Ice Cream Maker]";
		mes "Caro cliente, sua carteira parece estar leve.";
		mes "O preço é ^3355FF100 Zenys^000000 por sorvete.";
		close;
	}
	if(checkweight(536,.@input) == 0) {
		mes "[Ice Cream Maker]";
		mes "Caro cliente, parece que vocę está carregando muita coisa.";
		mes "Sorvete é bom,";
		mes "mas vocę deve considerar seu peso";
		mes "antes de efetuar uma compra.";
		close;
	}
	set Zeny,Zeny-100*.@input;
	getitem 536,.@input;	// Ice_Cream
	close;
}

// Alberta
