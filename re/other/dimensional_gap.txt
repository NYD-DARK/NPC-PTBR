dali,112,95,4	script	Logistics Manager	4_M_MERCAT2,{
	mes "[Logistics Manager]";
	mes "Bem-vindo. Estou encarregado da logística para cuidar de seus itens neste lugar estranho. Vocę precisa acessar seu armazém?";
	next;
	switch(select("Abrir Armazém 200 Zeny.:Năo, obrigado.")) {
	case 1:
		if (Zeny > 199) {
			if(!callfunc("F_CanOpenStorage")){
				mes "[Logistics Manager]";
				mes "Vocę ainda năo está qualificado para usar o Armazém.";
			}
			else {
				Zeny -= 200;
				openstorage;
			}
		}
		else {
			mes "[Logistics Manager]";
			mes "Vocę precisa de pelo menos 200 zeny para usar o Armazém.";
		}
		break;
	case 2:
		mes "[Logistics Manager]";
		mes "Deixe-me saber se mudar de ideia.";
	}
	close;
}
