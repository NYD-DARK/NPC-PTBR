prontera,82,108,5	script	Heart Merchant#life01	4_M_NFDEADMAN2,5,5,{
	.@old_item = 12209;
	.@new_item = 6413;
	.@old_item_name$ = getitemname(.@old_item);
	.@new_item_name$ = getitemname(.@new_item);

	if (!checkweight(1201,1) || (MaxWeight - Weight) < 500) {
		mes "Vocę năo pode prosseguir com a conversa porque possui uma grande quantidade de itens.";
		mes "Por favor, organize seus itens e tente novamente.";
		close;
	}
	mes "[Heart Merchant]";
	mes "Olá! Vejo que vocę é um aventureiro que entrega seu coraçăo ŕs viagens, usa um barco para navegar pelo continente e aproveita a emoçăo do perigo!";
	next;
	if(!countitem(.@old_item)){
		mes "[Heart Merchant]";
		mes "Vocę tem algum" + mesitemlink( .@old_item ) + "no seu armazém?";
		next;
		mes "[Heart Merchant]";
		mes "Se vocę me trouxer um" + .@old_item_name$ + ", eu trocarei por" + mesitemlink( .@new_item ) + ". O que vocę acha? Vocę tem ideia do que é este item?";
	} else {
		mes "[Heart Merchant]";
		mes "Vocę já pensou em substituir seu" + mesitemlink( .@old_item ) + "por algo melhor? Por exemplo..." + mesitemlink( .@new_item ) + ". O que vocę acha?";
	}
	next;
	switch(select("O que é isso?",(countitem(.@old_item)?"Troque, por favor!":""),"Năo estou interessado.")) {
	case 1:
		mes "[Heart Merchant]";
		mes mesitemlink( .@old_item ) + "é um item que evita que vocę perca experięncia ao morrer uma vez, dentro da duraçăo de 30 minutos após o uso.";
		next;
		mes "[Heart Merchant]";
		mes "Mas muitas pessoas ficam tristes com o limite de tempo, năo é? Entăo eu analisei isso e preparei o" + mesitemlink( .@new_item ) + ".";
		next;
		mes "[Heart Merchant]";
		mes mesitemlink( .@new_item ) + "é um item que evitará a perda de experięncia ao morrer uma vez, apenas por tę-lo ^EE0000independentemente da duraçăo^000000. Claro, após a morte, o"+.@new_item_name$+"é consumido.";
		next;
		select("Por que vocę está trocando isso?");
		mes "[Heart Merchant]";
		mes "Está curioso sobre isso? Eu sou um Mercador. Comprar e vender coisas é o meu destino, seja relacionado ŕ vida ou ŕ morte de alguém. Vocę acha que eu vou perder esta oportunidade de ganhar dinheiro? Eu ganho dinheiro, e os vivos obtęm um alívio da dor da morte. Năo é um bom negócio?";
		next;
		mes "[Heart Merchant]";
		mes "A taxa de troca é de 5" + mesitemlink( .@old_item ) + "por 1" + mesitemlink( .@new_item ) + ". Vocę tem alguma dúvida?";
		next;
		mes "[Heart Merchant]";
		mes "Hm~ O que vocę acha? Năo é uma troca justa? Vocę quer sempre se preocupar com o tempo? É um item indispensável para aventureiros que desfrutam de aventuras de tirar o fôlego.";
		close;
		
	case 2:	
		mes "[Heart Merchant]";
		mes "Sim, sim. Boa decisăo. É melhor trocar por um melhor, năo é?";
		next;
		mes "[Heart Merchant]";
		mes "Eu posso trocar seus ^0000cd5"+.@old_item_name$+"por 1"+.@new_item_name$+".";
		next;
		mes "[Heart Merchant]";
		mes "No entanto, ^EE0000[Năo ŕ venda]" + .@old_item_name$ + "năo é elegível para troca^000000, entăo é inútil trazę-lo aqui e pedir para trocá-lo.";
		next;
		mes "[Heart Merchant]";
		mes "Vamos fazer a troca agora?";
		next;
		if (select("Troque, por favor!:Eu năo quero fazer a troca.") == 2)
			break;
		mes "[Heart Merchant]";
		mes "Quantos vocę deseja trocar? ^0000cdVocę pode trocar até 10"+.@new_item_name$+"por vez^000000.";
		next;
		input .@amount;
		if (.@amount == 0) {
			mes "[Heart Merchant]";
			mes "Hm? Vocę năo está interessado? Este é um ótimo produto e é muito bom para alguém como vocę. Mas se vocę năo quer proteger sua vida, eu também năo vou forçá-lo.";
			close;
		}
		if (.@amount > 10) {
			mes "[Heart Merchant]";
			mes "Como eu disse, ^0000cdvocę só pode trocar até 10^000000 por vez. Verifique a quantidade novamente.";
			close;
		}
		if (countitem(.@old_item) < (.@amount*5)) {
			mes "[Heart Merchant]";
			mes "Hum, com licença, mas... vocę năo tem" +.@old_item_name$+ "suficientes, năo é? Săo 5" + mesitemlink( .@old_item ) + "para 1" + mesitemlink( .@new_item ) + ". Por favor, verifique a quantidade.";
			close;
		}
		mes "[Heart Merchant]";
		mes "Agora, com isso, eu troquei seu ^0000cd" + .@old_item_name$ + "^000000 pelo ^EE0000" + .@new_item_name$ + "^000000!! Esse foi um bom negócio, năo foi? Na próxima vez que vocę colocar as măos em um"+.@old_item_name$+", venha me ver novamente!";
		delitem .@old_item,.@amount*5;
		getitem .@new_item,.@amount;
		close;
	case 3:
		break;
	}
	mes "[Heart Merchant]";
	mes "Este é um ótimo negócio! Aventureiro, é um item indispensável para pessoas como vocę que se arriscam em lugares perigosos!";
	next;
	mes "[Heart Merchant]";
	mes "Se vocę ficar interessado nele, năo deixe de voltar para a troca. Ok?";
	close;
	
OnTouch:
	if(countitem(12209))
		npctalk "Vocę tem algo bom. Gostaria de substituir seu" + getitemname(12209) + " with a better one?", "", bc_self;
	end;
}
