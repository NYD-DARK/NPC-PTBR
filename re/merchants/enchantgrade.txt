grademk,34,184,4	script	Sratos#sratos	4_JP_GARM_H,{
	mes "[Sratos]";
	mes "Olá, querido cliente que entrou, vamos torcer por um milagre hoje.";
	mes "Como posso te ajudar?";
	next;
	switch( select( "Aprimorar o nível do equipamento.", "Trocar itens de Etel", "O que é o Aprimoramento de Nível?" ) ){
		case 1:
			mes "[Sratos]";
			mes "Nosso cliente. Você quer aprimorar o nível do equipamento.";
			mes "Não é fácil mover o poder mágico das joias.";
			next;
			mes "[Sratos]";
			mes "O poder mágico pode explodir no local.";
			mes "Então a arma que herdará o poder mágico também explode!";
			next;
			switch( select( "Eu ainda quero fazer isso!", "Vou pensar a respeito." ) ){
				case 1:
					mes "[Sratos]";
					mes "Desejo boa sorte aos nossos corajosos clientes!";
					close2;
					enchantgradeui();
					end;
				case 2:
					mes "[Sratos]";
					mes "Sempre que tiver a coragem de desafiar, por favor, volte.";
					close;
			}
		case 2:
			mes "[Sratos]";
			mes "Estas são joias usadas para aprimorar níveis.";
			mes "Pó de Etel, as joias que serão a base e, se você me der uma pequena taxa, eu trocarei por joias de Etel.";
			close2;
			callshop "EnchantGradeExchange";
			end;

		case 3:
			mes "[Sratos]";
			mes "Ocasionalmente, joias instáveis com muito poder mágico puro são encontradas.";
			mes "Nós chamamos essas de gemas etéricas, certo?";
			next;
			mes "[Sratos]";
			mes "Você pode transferir o poder mágico da gema etérica para outras armas, mas é claro que há algum risco.";
			mes "É delicado, então se você não tiver cuidado, puf! Vai explodir.";
			next;
			mes "[Sratos]";
			mes "De qualquer forma, se você usar esta gema etérica, você pode aprimorar a arma que está usando.";
			mes "O desempenho melhora. Chamamos isso de aprimoramento de nível.";
			next;
			mes "[Sratos]";
			mes "Você também, não ficaria feliz se as armas que ama crescessem um passo a mais?";
			mes "Nós somos as pessoas que auxiliam nisso. É um pouco arriscado, no entanto. ahahaha.";
			close;
	}

OnInit:
	setunittitle(getnpcid(0), "<Grade Enhancer>");
	end;
}
