dali,139,118,4	script	Dimensional Device#sara	10007,{
	if (BaseLevel < 99) {
		mes "[Adventurer Leon]";
		mes "Ei! Vocę é muito jovem para brincar com isso! Volte quando tiver pelo menos o ^0000FFNível 99^000000!";
		close;
	}
	if ((saramemory < 1) || (saramemory == 2) || (saramemory == 3) || (saramemory == 5)) {
		mes "[Adventurer Leon]";
		mes "Before you go playing around with that; Why don't you come over here and talk to me?";
		close;
	}

	// 20 hour cooldown
	set .@ins_sara_time,checkquest(15002,PLAYTIME);
	if (.@ins_sara_time == -1) {

		set .@party_id,getcharid(1);
		set .@p_name$,getpartyname(.@party_id);
		set .@md_name$,"Sara's Memories";

		if (!.@party_id) {
			mes "[Adventurer Leon]";
			mes "Antes de entrar, vocę precisa organizar um grupo! Năo sabe como? Digite /organize NOMEDOGRUPO.";
			close;
		}

		if (is_party_leader() == true)
			set .@menu$, "Inicializar o dispositivo dimensional.:Usar o dispositivo dimensional.:Cancelar.";
		else
			set .@menu$, ":Use o dispositivo dimensional.:Cancelar.";

		switch(select(.@menu$)) {
		case 1:
			if (instance_create(.@md_name$) < 0) {
				mes "Nome do Grupo:"+.@p_name$;
				mes "Líder do Grupo:"+strcharinfo(0);
				mes "^0000ff"+.@md_name$+"^000000- Reserva falhou!";
				close;
			}
			mes "^FF0000O dispositivo dimensional inicializa perfeitamente. Use o dispositivo para entrar na fenda no espaço e tempo.^000000";
			close;
		case 2:
			switch(instance_enter(.@md_name$)) {
			case IE_OTHER:
				mes "Um erro desconhecido ocorreu.";
				close;
			case IE_NOINSTANCE:
				mes "A instância memorial"+.@md_name$+"năo existe.";
				mes "O líder do grupo ainda năo gerou a instância.";
				close;
			case IE_NOMEMBER:
				mes "Apenas os membros registrados podem entrar na instância"+.@md_name$+".";
				close;
			case IE_OK:
				mapannounce "dali",strcharinfo(0)+", membro do grupo"+.@p_name$+" entered the instance "+.@md_name$+".",bc_map,"0x00ff99";
				setquest 15002;
				end;
			}
		case 3:
			mes "^FF0000O líder do grupo năo inicializou o dispositivo dimensional.^000000";
			close;
		}
	} else if (.@ins_sara_time == 0 || .@ins_sara_time == 1) {
		mes "[Adventurer Leon]";
		mes "Acho que, cerca de 20 horas após sua última entrada na fenda, ela estará estável o suficiente para vocę atravessá-la novamente.";
		close;
	}
	else {
		erasequest 15002;
		mes "^0000FF[ Memórias de Sara ] O tempo de espera foi reiniciado. Agora é possível usar o dispositivo dimensional novamente.^000000";
		close;
	}
}
