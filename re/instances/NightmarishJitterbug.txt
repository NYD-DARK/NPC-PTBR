moc_para01,29,95,5	script	Runane#sara	4_F_RUNAIN,{
	cutin "lunain01",0;
	if (BaseLevel < 120) {
		mes "[Runane]";
		mes "^ff0000Você precisa estar no Nível 120.^000000";
		close3;
	}
	if (isbegin_quest(13181) != 1) {
		mes "[Runane]";
		mes "^ff0000Se você quer usar meu portal dimensional, primeiro precisa falar com o Roel. O portal não está aberto para todos.^000000";
		close3;
	}
	.@party_id = getcharid(1);
	.@md_name$ = "Nightmarish Jitterbug";
	if (.@party_id < 1) {
		mes "[Runane]";
		mes "^ff0000Você primeiro deve formar/entrar em um grupo de pelo menos 1 pessoa para entrar nesta masmorra.^000000";
		close3;
	}
	switch( checkquest(13182,PLAYTIME) ) {
	case -1:
		break;
	case 0:
	case 1:
		mes "[Runane]";
		mes "^ff0000Este portal dimensional não pode ser acessado por 20 horas após o seu último acesso.^000000";
		close3;
	case 2:
		mes "[Runane]";
		mes "O portal está disponível para uso agora.";
		erasequest 13182;// Nightmarish Jitterbug: Waiting
		close3;
	}
	if (is_party_leader() == true)
		.@menu$ = "Abrir o portal dimensional.";
	switch( select( .@menu$, "Entrar" + .@md_name$ + ".", "Cancelar." ) ) {
	case 1:
		if (instance_create(.@md_name$) < 0) {
			mes "Nome do Grupo:" + getpartyname( getcharid(1) );
			mes "Líder do Grupo:" + strcharinfo(0);
			mes "^0000ff" + .@md_name$ + "^000000- Reserva Falhou!";
			close3;
		}
		mes "[Runane]";
		mes "^ff0000Preparando o portal dimensional...^000000";
		mes "^ff0000Por favor, aguarde.^000000";
		close3;
	case 2:
		switch( instance_enter(.@md_name$) ) {
		case IE_NOMEMBER:
			close3;
		case IE_NOINSTANCE:
			if (is_party_leader() == false) {
				mes "[Runane]";
				mes "Teletransportando...";
				next;
				mes "[Runane]";
				mes "^ff0000Observe que este portal não pode ser acessado por 23 horas após o seu último acesso.^000000";
				next;
			}
			mes "[Runane]";
			mes "^ff0000O líder do seu grupo ainda não criou uma Fenda Temporal.^000000";
			close3;
		case IE_OTHER:
			mes "Ocorreu um erro desconhecido.";
			close3;
		case IE_OK:
			setquest 13182;// Nightmarish Jitterbug: Waiting
			mapannounce "moc_para01", "" + strcharinfo(0) + "do grupo," + getpartyname( getcharid(1) ) + ", is entering the " + .@md_name$ + ".", bc_map, 0xFF99;
			specialeffect2 EF_READYPORTAL2;
			specialeffect2 EF_PORTAL2;
			// warp "1@jtb",16,17;
			end;
		}
	case 3:
		cutin "",255;
		end;
	}
}
