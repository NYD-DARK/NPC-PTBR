// Illusion of Luanda
//==================================================
// Todo: the count should start after the end of the cooldown?
-	script	illusion_luanda	-1,{
	end;
OnMob20270: callsub( S_Mob, 20270, 5 );	// ILL_TRI_JOINT
OnMob20271: callsub( S_Mob, 20271, 5 );	// ILL_STALACTIC_GOLEM
OnMob20272: callsub( S_Mob, 20272, 5 );	// ILL_MEGALITH
OnMob20274: callsub( S_Mob, 20274, 4 );	// ILL_STONE_SHOOTER
OnMob20275: callsub( S_Mob, 20275, 4 );	// ILL_WOOTAN_SHOOTER
OnMob20276: callsub( S_Mob, 20276, 4 );	// ILL_WOOTAN_FIGHTER
S_Mob:
	.@group = getarg(1);
	if (.@group != .group)
		end;
	if (playerattached()) {
		if (.count < 1500)
			.count++;
		if (.count == 1500 && !.boss_spawn && !.cooldown) {
			getunitdata( killedgid, .@data );
			callsub( S_Boss, (20273 + .swap), .@data[UMOB_X], .@data[UMOB_Y] );
		}
	}
	.@mob_id = getarg(0);
	sleep 5000;
	if (.@group == .group)
		monster "com_d02_i",0,0,"--ja--",.@mob_id,1,"illusion_luanda::OnMob" + .@mob_id;
	end;

S_Boss:
	killmonster "com_d02_i", "illusion_luanda::OnBossDeath";
	.boss_spawn = true;
	monster "com_d02_i",getarg(1),getarg(2),"--ja--",getarg(0),1,"illusion_luanda::OnBossDeath";
	if (.swap == 0)	// ILL_TAO_GUNKA
		mapannounce "com_d02_i", "Tao Gunka Ancestral: É hora de mostrar seu poder megalítico?", bc_map, 0x70DBDB;
	else	// ILL_WOOTAN_DEFENDER
		mapannounce "com_d02_i", "Defensor Wootan Ancestral: Acho que terei que intervir depois de todos esses inúteis.", bc_map, 0x70DBDB;
	end;

OnBossDeath:
	killmonster "com_d02_i", "illusion_luanda::OnBossDeath";
	.boss_spawn = false;
	.cooldown = true;

	if (.swap == 0) {	// ILL_TAO_GUNKA
		mapannounce "com_d02_i", "Tao Gunka Ancestral: Megalitos, é hora de se agachar e prender a respiraçăo.", bc_map, 0x70DBDB;
		sleep 2000;
		mapannounce "com_d02_i", "Tao Gunka Ancestral: Ficaremos mais fortes conforme nos soltarmos e caminharmos para o mundo.", bc_map, 0x70DBDB;
		sleep 2000;
		mapannounce "com_d02_i", "Tao Gunka Ancestral: Agachem-se. Quando vocęs se agacham, vocęs se tornam megalíticos.", bc_map, 0x70DBDB;
	}
	else {	// ILL_WOOTAN_DEFENDER
		mapannounce "com_d02_i", "Defensor Wootan Ancestral: Urgh... Um ferimento bastante perigoso...", bc_map, 0x70DBDB;
		sleep 2000;
		mapannounce "com_d02_i", "Defensor Wootan Ancestral: Vocę me forçou, um guerreiro da tribo Wootan, até este ponto. É muito bom. É uma pena, mas vou me retirar agora.", bc_map, 0x70DBDB;
		sleep 2000;
		mapannounce "com_d02_i", "Defensor Wootan Ancestral: Escutem, todos os Wootans! Saiam agora e preparem-se para o próximo.", bc_map, 0x70DBDB;
	}
	sleep 1000;
	.group = (.group == 4 ? 5 : 4);

	// all the mobs are killed and immediately replaced with new mobs
	for ( .@i = 0; .@i < 3; ++.@i ) {
		.@mob_id = 20270 + .@i + .swap;
		killmonster "com_d02_i", "illusion_luanda::OnMob" + .@mob_id;
	}

	.swap = .swap ^ 4;	// swap 0 -> 4, 4 -> 0 (used for mob ID)

	callsub( S_SpawnInit,
		(20270 + .swap),
		(20271 + .swap),
		(20272 + .swap)
	);
	initnpctimer;
	end;

OnTimer7200000:
	.cooldown = false;
	stopnpctimer;
	end;

S_SpawnInit:
	.count = 0;
	monster "com_d02_i",0,0,"--ja--",getarg(0),50,"illusion_luanda::OnMob" + getarg(0);
	monster "com_d02_i",0,0,"--ja--",getarg(1),50,"illusion_luanda::OnMob" + getarg(1);
	monster "com_d02_i",0,0,"--ja--",getarg(2),50,"illusion_luanda::OnMob" + getarg(2);
	return;

OnInit:
	.group = 5;	// UMOB_GROUP_ID

	callsub( S_SpawnInit,
		20270,	// ILL_TRI_JOINT
		20271,	// ILL_STALACTIC_GOLEM
		20272	// ILL_MEGALITH
	);
	end;
}
