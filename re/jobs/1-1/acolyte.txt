				mes "Bem, dou-lhe as boas-vindas";
				mes "de volta de Valhalla e";
				mes "desejo-lhe sorte em sua";
				mes "nova jornada de vida.";
				next;
				skill "NV_TRICKDEAD",0,SKILL_PERM;
				jobchange Job_Acolyte_High;
				skill "AL_HOLYLIGHT",1,SKILL_PERM;

				mes "Agora, aventure-se e busque aqueles que precisam de sua ajuda. Que Deus ilumine seu caminho.";
				close;
			}else{
				mes "[Father Mareusis]";
				mes "Agora, aventure-se para buscar pessoas que precisam de sua ajuda. Que Deus ilumine o seu caminhar.";
				close;
			}
		}else{
			mes "[Father Mareusis]";
			mes "Sinto que vocę suportou uma experięncia de vida passada. Vocę deve ter aprendido muitas coisas antes de entrar em Valhalla.";
			next;
			mes "[Father Mareusis]";
			mes "No entanto, posso dizer que vocę năo é adequado para ser um Noviço. Por favor, lembre-se de quem vocę foi em sua vida passada e encontre seu caminho.";
			close;
		}
	}
	mes "[Father Mareusis]";
	mes "O que é que vocę procura?";
	next;
	switch(select("Mudar sua classe para Noviço.:Perguntar os requisitos para ser um Noviço.:Sair.")) {
	case 1:
		mes "[Father Mareusis]";
		if(BaseJob != Job_Novice) {
			if(BaseJob == Job_Acolyte)
				mes "Vocę está se sentindo bem hoje? Posso dizer pelo seu traje que vocę já é um Noviço. Vocę năo está brincando, está?";
			else
				mes "Sinto muito, mas parece que vocę já tem sua própria classe, năo é?";
			close;
		}else{
			mes "Vocę deseja verdadeiramente se tornar um servo de Deus?";
			mes "Vamos ver se vocę está pronto para isso ou năo... Hmm...";
			next;
			if (!callfunc("F_CanChangeJob")) {
				mes "[Father Mareusis]";
				mes "Ora, puxa?! Vocę ainda năo completou o treinamento básico?! Vocę tem um longo caminho pela frente! Volte após aumentar seu nível de classe!";
				close;
			}
			mes "[Father Mareusis]";
			mes "Hmm... seu nível de classe é suficiente...";
			mes "Bom. Agora lhe darei a qualificaçăo para se tornar um Noviço.";
			next;
			mes "[Father Mareusis]";
			mes "Lembre-se sempre de ser grato a Deus, que cuida de nós o tempo todo. No caos e em tempos de dificuldade, enfrente suas provaçőes com fé inabalável.";
			callfunc "Job_Change",Job_Acolyte;
			getitem 1545,1;	// N_Mace
			next;
			mes "[Father Mareusis]";
			mes "Por último, quero sinceramente parabenizá-lo por perseverar em sua provaçăo de penitęncia.";
			close;
		}
	case 2:
		mes "[Father Mareusis]";
		mes "Vocę deseja se tornar um Noviço?";
		mes "Entăo, vocę deve cumprir os seguintes requisitos, pensando neles como as práticas dadas por Deus.";
		next;
		mes "[Father Mareusis]";
		mes "Primeiro, vocę deve alcançar pelo menos o Nível de Classe 9 de Aprendiz e aprender todas as Habilidades Básicas.";
		mes "Esta é a coisa mais básica a se fazer, entăo vocę precisa encará-la como uma forma de treinamento pessoal.";
		next;
		mes "[Father Mareusis]";
		mes "Quando achar que cumpriu este requisito, volte a mim novamente. Entăo vocę terá uma classe sagrada na qual poderá espalhar a vontade de Deus.";
		close;
	case 3:
		close;
	}
}
