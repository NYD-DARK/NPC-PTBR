function	script	cute_pet_manager	{
	// getarg(0) -> .@tame_id Array
	// getarg(1) -> .@tame_amount Array
	// getarg(2) -> .@hunt_id Array
	// getarg(3) -> .@hunt_amount Array
	// getarg(4) -> .@mob_id Array
	// getarg(5) ->  Attempt Variable
	// getarg(6) -> .@tame_gets Variable
	// getarg(7) -> .@hunt_id2 Array
	// getarg(8) -> .@hunt_amount2 Array

	mes "[Cute Pet Manager]";
	mes "Olá~! Eu sou um Gerente de Bichinhos";
	mes "responsável pelas relaçőes públicas";
	mes "do novo sistema aprimorado de Bichinhos.";
	next;
	
	mes "[Cute Pet Manager]";
	mes "Se vocę tiver alguma dúvida sobre";
	mes "o sistema de Bichinhos, estou aqui para te guiar.";
	mes "Diga-me qual item de domar";
	mes "vocę deseja,";
	mes "e eu lhe direi qual monstro";
	mes "vocę pode domar com esse item.";
	next;
	
	mes "[Cute Pet Manager]";
	mes "Também explicarei quais materiais";
	mes "vocę precisa trazer para";
	mes "conseguir o item de domar.";
	next;
	
	mes "[Cute Pet Manager]";
	mes "Se vocę já trouxe os materiais";
	mes "para trocar pelo item de domar,";
	mes "eu posso trocá-los por itens de domar imediatamente.";
	next;
	
	mes "[Cute Pet Manager]";
	mes "No entanto, existe um limite";
	mes "para a troca de itens de domar.";
	mes "Vocę só pode obter itens de domar";
	mes getarg(6) + "vezes. Entăo, por favor, certifique-se";
	mes "de quantas vezes vocę já trocou.";
	next;
	
	mes "[Cute Pet Manager]";
	mes "Entăo, qual item de domar vocę deseja?";
	mes "Por favor, escolha um da lista.";
	next;
	
	// Create Menu System
	set .@size, getarraysize(getarg(0));
	for (set .@a, 0; .@a < .@size; set .@a, .@a + 1)
		set .@menu$, .@menu$ + (.@menu$ == "" ? "" : ":") + getitemname(getelementofarray(getarg(0), .@a));
	
	// Query Player Choice
	set .@choice, select(.@menu$) - 1;

	// Store Variables (Less Lookup)
	set .@tame_id, getelementofarray(getarg(0), .@choice);
	set .@tame_amount, getelementofarray(getarg(1), .@choice);
	set .@hunt_id, getelementofarray(getarg(2), .@choice);
	set .@hunt_amount, getelementofarray(getarg(3), .@choice);
	set .@mob_id, getelementofarray(getarg(4), .@choice);
	set .@hunt_id2, getelementofarray(getarg(7), .@choice);
	set .@hunt_amount2, getelementofarray(getarg(8), .@choice);

	// dispbottom "Tame ID: " + .@tame_id;
	// dispbottom "Tame Amount: " + .@tame_amount;
	// dispbottom "Hunt ID: " + .@hunt_id;
	// dispbottom "Hunt Name: " + getitemname(.@hunt_id);
	// dispbottom "Hunt Amount: " + .@hunt_amount;
	// dispbottom "Mob ID: " + .@mob_id;
	if (.@hunt_id2) { 
		// dispbottom "Hunt ID2: " + .@hunt_id2;
		// dispbottom "Hunt Amount2: " + .@hunt_amount2;
		set .@hunt2_count, countitem(.@hunt_id2);
	}
	if (countitem(.@hunt_id) >= .@hunt_amount && countitem(6083) > 0 && .@hunt2_count >= .@hunt_amount2) {
		mes "[Cute Pet Manager]";
		mes "Ah, vocę reuniu todos os itens";
		mes "com seus amigos. Agora vocę pode";
		mes "trocar por um" + getitemname(.@tame_id) + ".";
		mes "Vocę deseja trocar comigo?";
		next;
		if  (select("Trocar:Don't Trocar") == 2) { 
			mes "[Cute Pet Manager]";
			mes "Năo quer? Vocę voltará novamente.";
			close;
		}
		delitem .@hunt_id, .@hunt_amount;
		delitem 6083, 1;
		if (.@hunt_id2)
			delitem .@hunt_id2, .@hunt_amount2;
		set getarg(5), getarg(5) + 1;
		getitem .@tame_id, .@tame_amount;
		
		mes "[Cute Pet Manager]";
		mes "Escolha sábia.";
		mes "Espero que vocę e seu bichinho se deem bem.";
		close;
	} else {
		mes "[Cute Pet Manager]";
		mes "Vocę escolheu" + getitemname(.@tame_id) + ".";
		mes "Vocę pode usá-lo para domar";
		mes "^FF0000" + getmonsterinfo(.@mob_id, MOB_NAME) + "^000000 monstros.";
		next;
		mes "[Cute Pet Manager]";
		mes "Traga ^FF0000" + .@hunt_amount + " " + getitemname(.@hunt_id) + "^000000" + (.@hunt_id2 ? ", ^FF0000" + .@hunt_amount2 + " " + getitemname(.@hunt_id2) : "") + "^000000 e ^FF00001 Cápsula de Boneca^000000";
		mes "e entăo vocę poderá trocá-los";
		mes "por um" + getitemname(.@tame_id) + ".";
		close;
	}
}
