ra_fild12,295,208,0	script	toairplane#rachel	45,1,1,{
	end;
OnTouch:
	mes "Para usar o Dirigível, vocę precisa pagar 1.200 zenys ou apresentar um Dirigível Ticket gratuito.";
	mes "Gostaria de utilizar o serviço?";
	next;
	if (select("Sim:Năo") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane_01",245,60;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane_01",245,60;
			end;
		}
		mes "Sinto muito, mas vocę năo tem dinheiro suficiente.";
		mes "Por favor, lembre-se de que é necessário pagar 1.200 zenys para usar o serviço.";
		close;
	}
	mes "Obrigado, volte sempre.";
	close;
}
