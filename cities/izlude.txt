-	script	::Sailor_izlude	-1,{
	mes "[Sailor]";
	mes "Ei, pessoal!";
	mes "Atençăo, atençăo!";
	mes "Venham navegar com o vento";
	mes "em um navio fascinante!";
	mes "Rápido, rápido!";
	next;
	switch(select("Ilha de Byalan -> 150 Zeny.:Marina de Alberta -> 500 Zeny.:Cancelar.")) {
	case 1:
		if (Zeny < 150) {
			mes "[Sailor]";
			mes "150 Zenys!";
			mes "Apenas 150 Zenys para viajar!";
			close;
		}
		set Zeny, Zeny - 150;
		warp "izlu2dun",107,50;
		end;
	case 2:
		if (Zeny < 500) {
			mes "[Sailor]";
			mes "500 Zenys!";
			mes "Apenas 500 Zenys para viajar!";
			close;
		}
		set Zeny, Zeny - 500;
		warp "alberta",188,169;
		end;
	case 3:
		close;
	}
}
