	mes "[Logistics]";

	if (!.@item_type)
		mes "Vocę selecionou ^ff0000"+ getitemname(.@item_id) +"^000000.";
	else if (.@item_type == 1)
		mes "Existe apenas uma"+ callfunc("F_GetWeaponType",.@item_id) +": "+ getitemname(.@item_id) +".";
	else if (.@item_type == 2) {

		mes "A armadura para alguém com nível acima de 80 é: ^ff0000"+ getitemname(.@item_id) +"^000000.";
	}
	else if (.@item_type == 3) {
		mes "Vocę selecionou 'Produçăo em Massa"+ callfunc("F_GetArmorType",.@item_id) +"'.";
		mes "Existe apenas uma Produçăo em Massa"+ callfunc("F_GetArmorType",.@item_id) +": ^ff0000"+ getitemname(.@item_id) +"^000000.";
	}
	else if (.@item_type == 4) {
		mes "Vocę selecionou 'Popularizada"+ callfunc("F_GetArmorType",.@item_id) +"'.";
		mes "Existe apenas uma Popularizada"+ callfunc("F_GetArmorType",.@item_id) +": ^ff0000"+ getitemname(.@item_id) +"^000000.";
	}

	if (!.@req_setting)
		mes "Requer ^0000ff2.000^000000 Pontos KVM, e"+ .@name$ +", seus pontos săo: ^580080"+ kvm_point +"^000000.";
	else if (.@req_setting == 4)

	else

	mes "Tem certeza de que deseja este item?";
	next;
	switch(select("Năo, năo vou comprar.:Sim, vou comprar.")) {
	case 1:
		mes "[Logistics]";
		mes "Vocę selecionou 'Năo vou comprar'.";
		mes "Ao comprar um item, lembre-se de que năo há reembolsos.";
