// First Tierra Gorge Officers - Guillaume
//============================================================
bat_room,124,178,5	script	Tierra Gorge Officer#01a	418,{
	if (getbattleflag("feature.bgqueue"))
		end;
	if (checkweight(1201,1) == 0) {
		mes "- Espere um minuto!! -";
		mes "- Vocę está carregando -";
		mes "- muitos itens. -";
		mes "- Por favor, tente novamente -";
		mes "- depois de perder um pouco de peso. -";
		close;
	}
	mes "[Guillaume Army Officer]";
	mes "Vamos mostrar o poder do Exército de Guillaume para aqueles Croixs fedorentos!";
	next;
	switch(select("Quero me juntar ao seu exército!:Encerrar Conversa")) {
	case 1:
		if ((BaseJob == Job_Novice) || (BaseJob == Job_Super_Novice)) {
			mes "[Guillaume Army Officer]";
			mes "Fico muito feliz que queira se juntar a nós para lutar contra o Exército Croix, mas sinto muito: năo posso enviar um novato como vocę para morrer no cruel campo de batalha.";
		}
		else {
			if (BaseLevel < 80) {
				mes "[Guillaume Army Officer]";
				mes "Fico muito feliz que queira se juntar a nós para lutar contra o Exército Croix, mas sinto muito: năo posso enviar um novato como vocę para morrer no cruel campo de batalha.";
			}
			else {
				set .@chk_urtime,checkquest(2069,PLAYTIME);
				if ((.@chk_urtime == 0)|| (.@chk_urtime == 1)) {
					mes "[Guillaume Army Officer]";
					mes "Vocę parece ter acabado de retornar do campo de batalha.";
					mes "É muito cedo para vocę voltar. Vá descansar e deixe os inimigos conosco!";
				}
				else {
					if (.@chk_urtime == 2)
						erasequest 2069;
					set .@mapcount,getmapusers("bat_a01");
					if (.@mapcount > 0) {
						mes "[Guillaume Army Officer]";
						mes "Recebi um relatório informando que um corpo de elite já foi enviado para o campo de batalha para lutar contra os Croixs.";
						mes "Por que vocę năo espera um pouco? Sugiro que afie suas armas e prepare seus suprimentos até lá.";
					}
					else {
						if ($@TierraBG1) {
							mes "[Guillaume Army Officer]";
							mes "Um corpo de elite já está a postos para ser enviado ao campo de batalha.";
							mes "Por que vocę năo espera um pouco? Sugiro que afie suas armas e prepare seus suprimentos até lá.";
						}
						else {
							mes "[Guillaume Army Officer]";
							mes "Vocę definitivamente parece estar pronto para a batalha!";
							mes "Vá mostrar aos Croixs o que o medo realmente significa!";
							mes "Hoje, nosso grito de vitória ecoará por todo o campo de batalha!";
							close2;
							warp "bat_room",57,223;
							end;
						}
					}
				}
			}
		}
		break;
	case 2:
		mes "[Guillaume Army Officer]";
		mes "Hoje, seremos vitoriosos!";
		break;
	}
	close;
}

bat_room,127,178,5	script	Guillaume Knight#1	417,{
	mes "[Guillaume Knight]";
	mes "A Garganta de Tierra consiste em dois lados íngremes colocados verticalmente e possui depósitos de raçőes para os Exércitos de Guillaume e Croix nas direçőes de 11 e 7 horas.";
	next;
	mes "[Guillaume Knight]";
	mes "A batalha começa no navio do seu exército, e o objetivo é avançar e destruir o depósito de raçőes do seu inimigo mais rápido do que eles podem destruir o seu.";
	next;
	mes "[Guillaume Knight]";
	mes "O curandeiro do campo de batalha ressuscitará os soldados uma vez a cada 25 segundos para que eles possam retornar ŕ batalha.";
	mes "O exército que capturar a bandeira neutra no centro do campo de batalha será recompensado com pontos extras de regeneraçăo, o que significa que seus soldados ressuscitarăo mais do que o outro lado, dando-lhes uma vantagem.";
	next;
	mes "[Guillaume Knight]";
	mes "Esses dois depósitos de raçőes estăo conectados entre si por caminhos curtos e estreitos, mas os portőes estăo bloqueados com barricadas.";
	mes "Pense e mova-se o mais rápido que puder. A vitória do seu exército depende da sua contribuiçăo.";
	next;
	mes "[Guillaume Knight]";
	if (getbattleflag("feature.bgqueue"))
		mes "Se estiver pronto, comece clicando no ^0000FFbotăo de batalha na interface^000000.";
	else {
		mes "Lembre-se: ao entrar em uma batalha, vocę receberá um token que indica a duraçăo definida durante a qual vocę năo poderá participar do mesmo tipo de batalha.";
		mes "Vocę pode verificar a duraçăo pressionando as teclas Alt+U.";
	}
	next;
	mes "[Guillaume Knight]";
	if (getbattleflag("feature.bgqueue"))
		mes "Por favor, lembre-se: uma vez que cancelar, ^0000FFvocę năo poderá se candidatar novamente por 1 minuto^000000!";
	else
		mes "Vocę está pronto para a batalha? Entăo se candidate com o recrutador ao meu lado!";
	close;
}
