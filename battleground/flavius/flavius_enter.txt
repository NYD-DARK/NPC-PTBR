//============================================================
bat_room,133,178,5	script	Flavius Officer#01a	418,{
	if (getbattleflag("feature.bgqueue"))
		end;
	if (checkweight(1201,1) == 0) {
		mes "- Espere um minuto !! -";
		mes "- Atualmente vocę está carregando -";
		mes "- itens demais. -";
		mes "- Por favor, tente novamente -";
		mes "- depois de perder um pouco de peso. -";
		close;
	}
	mes "[Guillaume Army Officer]";
	mes "Vamos mostrar o poder do Exército Guillaume para aqueles Croix fedorentos!";
	next;
	switch(select("Eu quero me juntar ao seu exército!:Encerrar Conversa")) {
	case 1:
		if ((BaseJob == Job_Novice) || (BaseJob == Job_Super_Novice)) {
			mes "[Guillaume Army Officer]";
			mes "Fico muito feliz que queira se juntar a nós e lutar contra o Exército Croix, mas sinto muito: năo posso enviar um novato como vocę para morrer no campo de batalha cruel.";
		}
		else {
			if (BaseLevel < 80) {
				mes "[Guillaume Army Officer]";
				mes "Fico muito feliz que queira se juntar a nós e lutar contra o Exército Croix, mas sinto muito: năo posso enviar um novato como vocę para morrer no campo de batalha cruel.";
			}
			else {
				set .@chk_urtime,checkquest(2070,PLAYTIME);
				if ((.@chk_urtime == 0)|| (.@chk_urtime == 1)) {
					mes "[Guillaume Army Officer]";
					mes "Parece que vocę acabou de retornar do campo de batalha.";
					mes "É muito cedo para vocę voltar. Vá descansar e deixe os inimigos conosco!";
				}
				else {
					if (.@chk_urtime == 2)
						erasequest 2070;
					set .@mapcount2,getmapusers("bat_b01");
					if (.@mapcount2 > 0) {
						mes "[Guillaume Army Officer]";
						mes "Recebi um relatório informando que um corpo de elite já foi enviado para o campo de batalha lutando contra os Croix.";
						mes "Por que vocę năo espera um pouco? Sugiro que afie suas armas e prepare seus suprimentos até lá.";
					}
					else {
						if ($@FlaviusBG1 > 0) {
							mes "[Guillaume Army Officer]";
							mes "Um corpo de elite já está de prontidăo para ser enviado ao campo de batalha.";
							mes "Por que vocę năo espera um pouco? Sugiro que afie suas armas e prepare seus suprimentos até lá.";
						}
						else {
							mes "[Guillaume Army Officer]";
							mes "Vocę definitivamente parece pronto para a batalha!";
							mes "Vá mostrar aos Croix o que o medo realmente significa!";
							mes "Hoje, nosso grito de vitória ecoará por todo o campo de batalha!";
							close2;
							warp "bat_room",85,223;
							end;
						}
					}
				}
			}
		}
		break;
	case 2:
		mes "[Guillaume Army Officer]";
		mes "Hoje, seremos vitoriosos!";
		break;
	}
	close;
}

bat_room,135,178,5	script	Guillaume Knight#3	417,{

	next;
	mes "[Guillaume Knight]";
	mes "Os cristais de ambos os exércitos săo protegidos por barricadas especiais que năo podem ser destruídas por ataques diretos.";
	next;
	mes "[Guillaume Knight]";
	mes "Mas eles podem ser removidos ao destruir os Guardiőes que protegem a base do exército inimigo.";
	next;
	mes "[Guillaume Knight]";
	mes "The crystals are immune to every type of skill; your physical attacks are the only choice for destroying your enemy's crystal.";
	next;
	mes "[Guillaume Knight]";
	if (getbattleflag("feature.bgqueue"))
		mes "Se vocę estiver pronto, comece clicando no ^0000FFbotăo de batalha na interface^000000.";
	else {
		mes "Lembre-se: ao entrar em uma batalha, vocę receberá um token que indica a duraçăo definida pela qual vocę năo poderá participar do mesmo tipo de batalha.";
		mes "Vocę pode verificar a duraçăo pressionando as teclas Alt+U.";
	}
	next;
	mes "[Guillaume Knight]";
	if (getbattleflag("feature.bgqueue"))
		mes "Tenha em mente que, uma vez que vocę cancelar, ^0000FFvocę năo poderá se inscrever novamente por 1 minuto^000000!";
	else
		mes "Vocę está pronto para a batalha? Entăo inscreva-se com o recrutador ao meu lado!";
	close;
}
