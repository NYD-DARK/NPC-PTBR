		next;
		mes "[Blacksmith]";
		mes "Pelo bem do nosso ofício, nossos rostos lindos, porém masculinos, săo colocados em risco...";
		next;
		mes "[Blacksmith]";
		mes "Mas com essa maravilha da tecnologia, năo há necessidade de se preocupar! Contemplem, a ^3355FFMáscara de Solda^000000 !!";
		next;
		mes "[Blacksmith]";
		mes "Ela funciona cobrindo seu rosto com uma grossa chapa de metal. Realmente, é um item que todos deveriam querer! Se năo para vocę, seria uma boa ideia dar isso a um amigo que esteja estudando a forja.";
		next;
		mes "[Blacksmith]";
		mes "Entăo, que tal? Se vocę me der ^2244FF50 Aços^000000 e ^4422FFapenas 2000 Zeny^000000, eu farei isso para vocę agora mesmo~!!";
		next;
		switch(select("Hm... Nada mal. Tudo bem.:Como ela funciona mesmo?")) {
		case 1:
			if ((countitem(999) > 49) && (Zeny > 1999)){
				mes "[Blacksmith]";
				mes "Ah, vejo que vocę aprecia o trabalho de um Ferreiro! Certo, farei isso agora mesmo!!";
				next;
				delitem 999,50;  //Steel
				set Zeny,Zeny-2000;

				mes "Hahaha~! Faça bom uso disso!";
				close;
			}
			else {
				mes "[Blacksmith]";
				mes "Hmmm...";
				mes "Vocę năo tem os itens que eu exijo.";
				next;
				mes "[Blacksmith]";
				mes "Um Ferreiro comum interpretaria isso como um sinal de desrespeito, mas eu vou perdoá-lo. Apenas năo se esqueça do que eu preciso se quiser que eu faça esta máscara para vocę.";
				close;
			}
		case 2:
			mes "[Blacksmith]";
			mes "A ^3355FFMáscara de Solda^000000 protege seu rosto cobrindo-o com uma grossa chapa de metal. Esta invençăo é o resultado de 30 anos de trabalho interminável no martelo e na bigorna.";
			close;
		}
	case 2:
		mes "[Blacksmith]";
		mes "Ah...!";
		mes "Acho que esta é a primeira vez que vocę conhece um Ferreiro tăo grandioso quanto eu, e vocę está sem palavras de admiraçăo. Por favor, leve o seu tempo e fale quando estiver pronto.";
		next;
		mes "^3355FFEste cara parece";
		mes "um pouco cheio";
		mes "de si.^000000";
		close;
	}
}

// Number Match Game :: gef_yagu
//============================================================
geffen,122,77,4	script	Nia#yagu	700,{
	mes "[Nia]";
	mes "Olá, sou Nia, a quinta";
	mes "discípula de Mawong, o";
	mes "mestre dos jogos. Agora,";
	mes "que tal jogar um jogo de";
	mes "Combinar Números comigo?";
	next;
	switch(select("Jogo de Combinar Números?:Praticar Combinar Números:Jogar Combinar Números:Recordista:Cancelar")) {
	case 1:
		mes "[Nia]";
		mes "Ah, bem, o objetivo do";
		mes "jogo de Combinar Números é";
		mes "adivinhar o número em que";
		mes "estou pensando. Claro, existem";
		mes "algumas regras que limitam os";
		mes "números que posso escolher.";
		next;
		mes "[Nia]";
		mes "Todos os números que eu crio";
		mes "devem ter 3 dígitos, năo posso";
		mes "usar 0 em nenhum dos dígitos,";
		mes "e năo posso usar o mesmo";
		mes "número em nenhum dos dígitos";
		mes "mais de uma vez.";
		next;
		mes "[Nia]";
		mes "Entăo, por exemplo, năo posso";
		mes "usar os números 103, 112,";
		mes "252 ou 701. Esses números";
		mes "possuem um 0 ou usam";
		mes "o mesmo número nos dígitos";
		mes "mais de uma vez. Entendeu agora?";
		next;
		mes "[Nia]";
		mes "Agora, vocę só tem 5 tentativas,";
		mes "e após cada tentativa, eu direi";
		mes "quantos dígitos vocę adivinhou";
		mes "corretamente, seguido pelo número de dígitos adivinhados corretamente na";
		mes "posiçăo correta.";
		next;
		mes "[Nia]";
		mes "Por exemplo, digamos que o";
		mes "número que eu criei seja 168.";
		mes "Se seu primeiro palpite for 678, entăo vocę acertou dois dos dígitos,";
		mes "6 e 8. No entanto,";
		mes "apenas o 8 está no lugar correto.";
		next;
		mes "[Nia]";
		mes "Sabe, seria uma boa ideia";
		mes "se vocę experimentasse o Modo Treino";
		mes "primeiro. Entăo, quando estiver mais";
		mes "confortável, jogue o jogo";
		mes "de Adivinhaçăo de Números de verdade. Năo esqueça que a velocidade conta para sua pontuaçăo!";
		next;
		mes "[Nia]";
		mes "Ah, se por acaso vocę";
		mes "fizer um novo recorde no";
		mes "jogo de Adivinhaçăo de Números,";
		mes "seu nome será registrado,";
		mes "entăo năo perca esta chance de";
		mes "fama e talvez até fortuna~";
		close;
	case 2:
		mes "[Nia]";
		mes "Ótimo, vamos fazer um treino";
		mes "de Adivinhaçăo de Números. Por favor,";
		mes "tente adivinhar meu número, tudo bem?";
		mes "Se quiser cancelar, digite um número maior que 1.000.";
		set .@practice,1;
		break;
	case 3:
		mes "[Nia]";
		mes "Ótimo, vamos jogar Adivinhaçăo de Números!";
		mes "Vocę terá 5 chances para adivinhar o número em que estou pensando.";
		mes "Se quiser cancelar, digite um número maior que 1.000.";
		set .@practice,0;
		break;
	case 4:
		set .@name_record$, $050908_minus1_yagu$;
		if (.@name_record$ == "") //custom
			set .@name_record$,"(null)";
		set .@score_record, $050908_minus1_yagu;
		set .@score_min, .@score_record / 60;
		set .@score_sec, .@score_record % 60;
		mes "[Nia]";
		mes "^ff0000" + .@name_record$ + "^000000 é o";
		mes "recordista do jogo de Adivinhaçăo";
		mes "de Números com um tempo de";
		mes "^ff0000" + .@score_min + "minutos," + .@score_sec + "segundos^000000.";
		close;
	case 5:
		mes "[Nia]";
		mes "Cuide-se bem,";
		mes "aventureiro. Se algum dia";
		mes "sentir vontade de jogar um";
		mes "Adivinhaçăo de Números, venha falar";
		mes "comigo, certo? Até mais~";
		close;
	}
	next;

	// Generate 3 digits of a number.
	while(1) {
		set .@yagu100, rand(1,9);
		set .@yagu10, rand(1,9);
		set .@yagu1, rand(1,9);
		if ((.@yagu100 != .@yagu10) && (.@yagu100 != .@yagu1) && (.@yagu10 != .@yagu1)) {
			// Store the actual number (unofficial).
			set .@yagu, (100*.@yagu100) + (10*.@yagu10) + .@yagu1;
			break;
		}
	}

	// The official time calculation is based on ServerTime,
	// an int HHMMSS equivalent to atoi(gettimestr("%H%M%S",7)).
	// UNIX time is really a better way to do this.
	set .@start_time, gettimetick(2);

	while(1) {

		// Guess a number that meets all conditions.
		while(1) {
			if (input(.@input,100,999) != 0) {
				mes "[Nia]";
				mes "Vocę cancelou o";
				mes "jogo de Adivinhaçăo de Números.";
				close;
			}
			set .@input100, .@input / 100;
			set .@input10, (.@input % 100) / 10;
			set .@input1, .@input % 10;
			if ((.@input100 > 0) && (.@input10 > 0) && (.@input1 > 0)) {
				if ((.@input100 != .@input10) && (.@input100 != .@input1) && (.@input10 != .@input1))
					break;
			}
			mes "[Nia]";
			mes "Vocę inseriu um dos";
			mes "dígitos como 0, ou inseriu o";
			mes "mesmo valor numérico";
			mes "nos dígitos mais de uma vez.";
			mes "Por favor, insira outro palpite.";
			next;
		}

		set .@retry, .@retry+1;
		mes "[Nia]";
		mes "^ff0000"+strcharinfo(0)+"^000000,";
		mes "seu palpite é ^0000ff"+.@input+"^000000.";
		mes "Me dę um momento para";
		mes "processar seus resultados.";
		next;

		// Correct digits in correct locations.
		set .@strike,0;
		if (.@yagu100 == .@input100)
			set .@strike, .@strike+1;
		if (.@yagu10 == .@input10)
			set .@strike, .@strike+1;
		if (.@yagu1 == .@input1)
			set .@strike, .@strike+1;

		// Correct digits in incorrect locations.
		set .@ball,0;
		if ((.@yagu100 == .@input10) || (.@yagu100 == .@input1))
			set .@ball, .@ball+1;
		if ((.@yagu10 == .@input100) || (.@yagu10 == .@input1))
			set .@ball, .@ball+1;
		if ((.@yagu1 == .@input100) || (.@yagu1 == .@input10))
			set .@ball, .@ball+1;

		if (.@strike == 3) {
			if (.@practice) {
				mes "[Nia]";
				mes "Sim, o número que";
				mes "eu pensei foi ^ff0000"+.@yagu+"^000000!";
				mes "Parabéns, vocę";
				mes "acabou de vencer este jogo";
				mes "de treino! Agora vocę deve estar";
				mes "pronto para o jogo de verdade~";
				close;
			} else {

				// Calculate final time and check against the previous record.
				set .@end_time, gettimetick(2);
				if (.@end_time < .@start_time)
					set .@end_time, .@end_time + 3600;
				set .@total_time, .@end_time - .@start_time;
				set .@total_min, .@total_time / 60;
				set .@total_sec, .@total_time % 60;
				set .@name_record$, $050908_minus1_yagu$;
				if (.@name_record$ == "") //custom
					set .@name_record$,"(null)";
				set .@score_record, $050908_minus1_yagu;
				set .@score_min, .@score_record / 60;
				set .@score_sec, .@score_record % 60;

				if (.@score_record >= .@total_time || !.@score_record) { //custom (second condition)
					set $050908_minus1_yagu$, strcharinfo(0);
					set $050908_minus1_yagu, .@total_time;
					mes "[Nia]";
					mes "Ótimo! Vocę adivinhou";
					mes "meu número, que era";
					mes "^ff0000" + .@yagu + "^000000. Vocę terminou em";
					mes "^ff0000" + .@total_min + "minutos," + .@total_sec + "segundos^000000,";
					mes "quebrando o recorde anterior";
					mes "de ^ff0000" + .@score_min + "minutos," + .@score_sec + "segundos^000000.";
					next;
					mes "[Nia]";
					mes "^ff0000" + .@name_record$ + "^000000";
					mes "estabeleceu aquele antigo recorde,";
					mes "mas agora vocę é a";
					mes "nova pessoa a ser batida no";
					mes "Jogo de Números. Parabéns";
					mes "por estabelecer um novo recorde!";
					close;
				}
				mes "[Nia]";
				mes "Bom trabalho, vocę adivinhou";
				mes "meu número! A resposta, é";
				mes "claro, era ^ff0000" + .@yagu + "^000000. Vocę conseguiu";
				mes "adivinhar este número corretamente";
				mes "em ^ff0000" + .@total_min + "minutos," + .@total_sec + "segundos^000000.";
				next;
				mes "[Nia]";
				mes "^ff0000" + .@name_record$ + "^000000 é";
				mes "o atual recordista de Adivinhaçăo";
				mes "de Números com um tempo";
				mes "de ^ff0000" + .@score_min + "minutos," + .@score_sec + "segundos^000000.";
				mes "Boa sorte, e espero que vocę";
				mes "consiga quebrar este recorde algum dia~";
				close;
			}
		} else {
			if (.@practice) {
				mes "[Nia]";
				mes "Seu palpite tem ^ff0000" + .@ball + "^000000 dos";
				mes "números corretos. ^ff0000" + .@strike + "^000000 dos";
				mes "dígitos no seu palpite possuem";
				mes "o número correto na";
				mes "posiçăo de dígito correta.";
				mes "Certo, tente novamente~";
				next;
			} else {
				mes "[Nia]";
				mes "Seu último palpite teve ^ff0000" + .@ball + "^000000";
				mes "dos números corretos,";
				mes "e ^ff0000" + .@strike + "^000000 dígitos no seu palpite";
				mes "tinham o número correto na";
				mes "posiçăo de dígito correta.";
				switch(.@retry) {
					case 1: mes "Vocę tem quatro chances restantes."; break;
					case 2: mes "Vocę tem tręs chances restantes."; break;
					case 3: mes "Vocę tem duas chances restantes."; break;
					case 4: mes "Vocę só tem uma chance restante..."; break;
					case 5: mes "A resposta era ^ff0000" + .@yagu + "^000000."; close;
				}
				next;
			}
		}
	}
}
