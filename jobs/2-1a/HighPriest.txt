valkyrie,44,42,5	script	High Priest#Valkyrie	60,{
	if (ADVJOB == 0 || Upper != 1) {
		set .@karma_d,rand(1,10);
		if (.@karma_d > 4) {
			mes "[High Priest]";
			mes "Parabéns.";
			mes "Honra aos guerreiros!";
			close;
		}
		mes "[High Priest]";
		mes "Pelo poder";
		mes "da santidade, que possamos";
		mes "encontrar paz, força";
		mes "e proteção. Livrai-nos";
		mes "das forças do mal...";
		close;
	}
	else if (ADVJOB == Job_High_Priest && Class == Job_Acolyte_High && JobLevel > 39) {
		mes "[High Priest]";
		mes "Nosso mundo precisa";
		mes "de pessoas de";
		mes "talento e convicção.";
		mes "Por favor, continue seu";
		mes "bom trabalho como um";
		mes "herói ainda maior da santidade...";
		next;
		mes "[High Priest]";
		mes "Você gostaria";
		mes "de se tornar um Sumo Sacerdote?";
		next;
		if (select("Não.:Sim.") == 1) {
			mes "[High Priest]";
			mes "Quando estiver pronto,";
			mes "sinta-se à vontade para voltar.";
			mes "Honra aos guerreiros!";
			close;
		}
		if (SkillPoint) {
			mes "[High Priest]";
			mes "Ainda é possível que você aprenda mais habilidades. Por favor, use";
			mes "todos os seus Pontos de Habilidade restantes antes de retornar a mim.";
			close;
		}
		jobchange Job_High_Priest;
		set ADVJOB,0;
		mes "[High Priest]";
		mes "Parabéns.";
		mes "Como um Sumo Sacerdote,";
		mes "espero que você guie";
		mes "outros pelo caminho";
		mes "da santidade...";
		close;
	}
	mes "[High Priest]";
	mes "Bem-vindo";
	mes "a Valhalla,";
	mes "o Salão da Honra.";
	next;
	mes "[High Priest]";
	mes "Por favor, sinta-se";
	mes "à vontade";
	mes "enquanto estiver aqui.";
	mes "Honra aos guerreiros!";
	close;
}
