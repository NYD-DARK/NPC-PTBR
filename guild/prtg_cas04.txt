// Guild Flags (Outside Castle)
//============================================================
-	script	Skoegul#1::OutsideFlagsPR4	722,{

	set .@GID, GetCastleData("prtg_cas04",1);

	if (.@GID == 0) {
		mes "[ Edict of the Divine Rune-Midgarts Kingdom ]";
		mes " ";
		mes "1. Seguindo a ordenança do Divino Reino de Rune-Midgard,";
		mes "Declaramos que";
		mes "năo há mestre formal neste castelo.";
		mes " ";
		mes "2. Ŕquele que conseguir";
		mes "superar todas as provaçőes";
		mes "e destruir o Emperium,";
		mes "o rei concederá a posse";
		mes "deste castelo.";
		close;
	}
	if (getcharid(2) == .@GID) {
		mes "[ Echoing Voice ]";
		mes "Bravos guerreiros...";
		mes "Desejam retornar ao seu lugar de honra?";
		next;
		if(select("Retornar ao castelo da guilda.:Sair.") == 1) {
			close2;
			if (getcharid(2) == GetCastleData("prtg_cas04",1)) {
					warp "prtg_cas04",258,247;
				end;
			}
			end;
		}
		close;
	}
	mes "[ Edict of the Divine Rune-Midgarts Kingdom ]";
	mes " ";
	mes "1. Seguindo a ordenança do Divino Reino de Rune-Midgard,";
	mes "aprovamos que este local está sob";
	mes "a posse privada da ^ff0000"+GetGuildName(.@GID)+"Guilda ^000000.";
	mes " ";
	mes "2. O Mestre da guilda ^ff0000"+GetGuildName(.@GID)+"da Guilda ^000000 é";
	mes "^ff0000"+GetGuildMaster(.@GID)+"^000000";
	mes "Se houver alguém que se oponha a isso,";
	mes "prove sua força e honra com uma lâmina de aço em măos.";
	close;

OnRecvCastlePt04:
	FlagEmblem GetCastleData("prtg_cas04",1);
	end;
}
