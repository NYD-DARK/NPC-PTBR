// Guild Flag (Outside Castle)
//============================================================
-	script	Repherion#1::OutsideFlagsG1	722,{

	set .@GID, GetCastleData("gefg_cas01",1);

	if (.@GID == 0) {
		mes "[ Edict of the Divine Rune-Midgarts Kingdom ]";
		mes " ";
		mes "1. Seguindo a ordenança do Divino Reino de Rune-Midgard,";
		mes "declaramos que";
		mes "năo há um mestre formal deste castelo.";
		mes " ";
		mes "2. Ŕquele que conseguir";
		mes "superar todas as provaçőes";
		mes "e destruir o Emperium,";
		mes "o rei concederá a posse";
		mes "deste castelo.";
		close;
	}
	if (getcharid(2) == .@GID) {
		mes "[ Echoing Voice ]";
		mes "Bravos aventureiros...";
		mes "Desejam retornar ao seu lugar de honra?";
		next;
		if(select("Retornar ao castelo da guilda.:Sair.") == 1) {
			close2;
			if (getcharid(2) == GetCastleData("gefg_cas01",1)) {
				warp "gefg_cas01",197,36;
				end;
			}
			end;
		}
		close;
	}
	mes "[ Edict of the Divine Rune-Midgarts Kingdom ]";
	mes " ";
	mes "1. Seguindo a ordenança do Divino Reino de Rune-Midgard,";
	mes "aprovamos que este local está sob";
	mes "a posse privada de ^ff0000"+GetGuildName(.@GID)+"da guilda ^000000.";
	mes " ";
	mes "2. O Mestre da guilda ^ff0000"+GetGuildName(.@GID)+"da guilda ^000000 é";
	mes "^ff0000"+GetGuildMaster(.@GID)+"^000000";
	mes "Se houver alguém que se oponha a isto,";
	mes "prove sua força e honra com uma lâmina de aço em suas măos.";
	close;

OnRecvCastleG01:
	FlagEmblem GetCastleData("gefg_cas01",1);
	end;
}
